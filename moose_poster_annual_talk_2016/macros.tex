\newcommand{\HEADING}[1]
{
    \tikz \path[] 
        % node {\includegraphics[scale=0.3]{./images/heading.png}}
        node [right] {\fontsize{2.5cm}{1cm}\selectfont
            \textcolor{black}{\hspace{1cm}#1}
        };
    \\
}
